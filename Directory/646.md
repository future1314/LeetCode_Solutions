class Solution {
    public int findLongestChain(int[][] pairs) {
        if(pairs.length==0) return 0;
        Arrays.sort(pairs, (a,b)->a[1]-b[1]);
        int prev = pairs[0][1]; //prev is the end of last pair
        int count = 1;
        for(int i =1; i<pairs.length; i++){
            if(pairs[i][0]>prev){
                count++;
                prev = pairs[i][1];
            }
        }
        return count;
    }
}
